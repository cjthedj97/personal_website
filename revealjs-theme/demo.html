              name: 'cj-saathoff',
              accent: '#1abc9c',
              plugins: ['highlight', 'notes', 'search', 'zoom', 'math', 'markdown']
            };

            function loadTheme(deck) {
              deck.configure({ hash: true, slideNumber: true });
              return theme;
            }
